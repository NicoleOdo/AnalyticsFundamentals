4. Create a new variable called “age_group” and group the ages into categories

AgeGroup <- lego_sales %>%
  mutate(age_group = case_when(
    age < 19 ~ "18 and under",
    between(age, 19, 25) ~ "19 - 25",
    between(age, 26, 35) ~ "26 - 35",
    between(age, 36, 50) ~ "36 - 50",
    age > 50 ~ "51 and over"
  ))
